\begin{columns}
    \column{0.5}
    \block{What ?}{
        Đề xuất một khung đánh giá sử dụng VGGT như nguồn thông tin hình học hỗ trợ (không phải chuẩn tuyệt đối) để:
        \begin{itemize}
            \item Phân tích cấu trúc 3D trên các quỹ đạo điểm dài hạn.
            \item Đề xuất và kiểm chứng 3 nhóm chỉ số: độ ổn định vị trí 3D, sai số chiếu lại dài hạn, độ trôi của tỉ lệ \& độ lệch.
            \item Thử nghiệm hiệu chỉnh Test-Time Adaptation tối ưu tham số affine theo VGGT để giảm sai lệch.
        \end{itemize}
    }

    \column{0.5}
    \block{Why ?}{
        \begin{itemize}
            \item Các mô hình độ sâu video hiện đại (như Video Depth Anything) ổn định về mặt hình ảnh nhưng \textbf{\textit{\color{alertColor}chưa nhất quán về mặt hình học}} (trôi tỉ lệ, biến dạng không gian 3D).
            \item Các chỉ số hiện có (AbsRel, RMSE, TAE) chủ yếu đo sai số trên từng khung hình hoặc cặp khung hình liền kề, \textbf{\textit{\color{alertColor}chưa phản ánh mức độ sai lệch tích lũy}} dài hạn.
            \item Cần một công cụ để phát hiện, giải thích và giảm sai lệch hình học khó nhận biết bằng mắt thường.
        \end{itemize}
    }
\end{columns}

\begin{columns}
    \column{0.33}
    \block{1. Giới thiệu \& Mục tiêu}{
        Mục tiêu là xây dựng quy trình sử dụng VGGT để đánh giá và hiệu chỉnh tính nhất quán hình học 3D của bản đồ độ sâu video.
        \begin{itemize}
            \item \textbf{Bài toán:} Ước lượng độ sâu nhất quán.
            \item \textbf{Ý tưởng:} Dùng thông tin camera, điểm 3D và điểm theo dõi từ VGGT như tín hiệu hỗ trợ.
        \end{itemize}

        \vspace{0.45cm}
        \begin{center}
            \includegraphics[width=0.96\linewidth]{assets/proposal/video_depth_anything_teaser.png}

            \vspace{0.25cm}
            {\small\color{muted}Figure 1. Ước lượng độ sâu ổn định theo thời gian (nguồn: Video Depth Anything).}
        \end{center}
    }

    \column{0.34}
    \block{2. Phương pháp \& Chỉ số}{
        \textbf{Quy trình:}
        \begin{itemize}
            \item \textbf{Căn chỉnh chung:} Ước lượng một cặp tham số (scale \& shift) cho toàn bộ đoạn video trong không gian nghịch đảo độ sâu.
            \item \textbf{Chỉ số đề xuất:}
            \begin{itemize}
                \item \textit{Độ ổn định vị trí 3D:} Đo mức biến dạng của cùng một điểm qua nhiều khung hình.
                \item \textit{Sai số chiếu lại dài hạn:} Phân tích độ lệch chiếu lại theo khoảng cách thời gian.
                \item \textit{Mức trôi:} Theo dõi sự thay đổi của tỉ lệ và độ lệch.
            \end{itemize}
        \end{itemize}

        \vspace{0.45cm}
        \begin{center}
            \includegraphics[width=0.96\linewidth]{assets/proposal/vggt_architecture.png}

            \vspace{0.25cm}
            {\small\color{muted}Figure 2. Mô hình VGGT cung cấp tín hiệu hình học đồng thời.}
        \end{center}
    }

    \column{0.33}
    \block{3. Hiệu chỉnh \& Kết quả}{
        \textbf{Test-Time Adaptation:}
        \begin{itemize}
            \item Hiệu chỉnh tham số affine $x'_t = a_t x_t + b_t$ dựa trên chỉ số VGGT, cố định trọng số mô hình.
            \item So sánh giao thức dùng VGGT và giao thức dùng dữ liệu chuẩn (TUM RGB-D, Sintel).
        \end{itemize}

        \vspace{0.35cm}
        \textbf{Kết quả mong đợi:}
        \begin{itemize}
            \item Tỉ lệ xếp đúng thứ tự > 80\% đối với các mức độ sai lệch kiểm soát.
            \item Giảm trung vị sai số hình học tham chiếu.
        \end{itemize}

        \end{center}
